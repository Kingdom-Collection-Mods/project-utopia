namespace = utopia_journal_prod
# Heavy Machinery
utopia_journal_prod.1 = {
	title = utopia_journal_prod.heavy_industrial_machinery_je.success 
	type = country_event
	event_image = { video = "factory_precision" }
	
	option = {
		name = utopia.yes
		default_option = yes
		trigger = {
			NOT = { has_technology_researched = heavy_machinery_era_digital }
			can_research = heavy_machinery_era_digital
		}
		add_technology_progress = { progress = 4150 technology = heavy_machinery_era_digital }
	}
	
	option = {
		name = utopia.no
		
		trigger = {
			NOT = { has_technology_researched = heavy_machinery_era_digital }
			can_research = heavy_machinery_era_digital
			country_rank = rank_value:great_power
		}
		add_technology_progress = { progress = 6000 technology = heavy_machinery_era_digital }
	}
}

utopia_journal_prod.2 = {
	title = utopia_journal_prod.heavy_industrial_machinery_je.fail 
	type = country_event
	event_image = { video = "factory_precision" }
	
	
		
	immediate = {
		random_scope_state = {
			limit = {
				has_building = building_motor_industry
				is_incorporated = yes
			}
			save_scope_as = heavy_industrial_machinery_je_state
		}
	}
	
	option = {
		name = utopia.yes
		default_option = yes
		scope:heavy_industrial_machinery_je_state = {
			create_building = {
				building = building_heavy_machinery_factory
 				level=1
 				reserves=1
				 subsidized = yes
			}
			create_building = {
				building = building_aluminum_factory
 				level=1
 				reserves=1
				subsidized = yes
			}
			remove_building = building_motor_industry
		}
	}

	option = {
		name = utopia.no
	}
}
