ut_humiliate_power_bloc_goal = {
	groups= {
		general
	}
	requires_approval = no
	
	unlocking_technologies = {
		union_new_mandates
		union_new_mandates_two
	}
	
	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
		is_in_power_bloc = yes
		is_power_bloc_leader = yes
	}
	
	potential = {
		scope:target_country = {
			is_power_bloc_leader = yes
		}
		OR = {
			scope:actor.power_bloc ?= { has_principle = principle:ut_principle_military_cooperation_1 }
			scope:actor.power_bloc ?= { has_principle = principle:ut_principle_military_cooperation_2 }
			scope:actor.power_bloc ?= { has_principle = principle:ut_principle_military_cooperation_3 }
			has_technology_researched = union_new_mandates_two
		}
	}
	
	possible = {
		scope:target_country = {
			has_modifier = country_humiliated
			has_truce_with = root
		}
	}
	
	accept_effect = {
		scope:target_country = {
			join_power_bloc = root
		}
	}

	ai = {
		evaluation_chance = {
			value = 0.01 # AI shouldn't volunteer into power blocs all that often
		}
		
		will_propose = {  
			would_accept_diplomatic_action = {
				actor = scope:target_country
				type = ut_humiliate_power_bloc_goal
			}
		}
		
		propose_score = {
			value = 5
		}
	}
}

ut_influence_new_dawn_major_power = {
	groups= {
		general
	}
	requires_approval = yes
	
	unlocking_technologies = {
		union_new_mandates_two
	}
	
	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
		country_rank >= rank_value:great_power
		ut_new_dawn_has_core_trigger = yes
	}
	
	potential = {
		scope:target_country = {
			is_subject = no
			country_rank = rank_value:major_power
			NOT = { has_journal_entry = je_ut_new_dawn }
			hidden_trigger = {
				NOT = { has_variable = je_ut_new_dawn_done }
			}
		}
	}
	
	possible = {
		scope:target_country = {
			custom_tooltip = {
				text = DIPLO_ACTION_RELATIONS_POOR_OR_LESS_NOT
				scope:target_country.relations:root > relations_threshold:poor
			}
			ut_service_workers_possible = yes
			pop_type_percent_country = {
				pop_type = service_workers
				percent > 0.01
			}
		}
	}
	
	accept_effect = {
		if = {
			limit = {
				has_modifier = je_ut_new_dawn_democrat_focus
			}
			scope:target_country = {
				add_modifier = je_ut_new_dawn_democrat_focus
			}
		}
		else_if = {
			limit = {
				has_modifier = je_ut_new_dawn_autocrat_focus
			}
			scope:target_country = {
				add_modifier = je_ut_new_dawn_autocrat_focus
			}
		}
		else_if = {
			limit = {
				has_modifier = je_ut_new_dawn_communist_focus
			}
			scope:target_country = {
				add_modifier = je_ut_new_dawn_communist_focus
			}
		}
		else_if = {
			limit = {
				has_modifier = je_ut_new_dawn_fascist_focus
			}
			scope:target_country = {
				add_modifier = je_ut_new_dawn_fascist_focus
			}
		}


		scope:target_country = {
			set_variable = je_ut_new_dawn_major_power_influenced_var
			add_modifier = {
				name = ut_new_dawn_focus_recently_changed
				days = normal_modifier_time
			}
			add_journal_entry = {
				type = je_ut_new_dawn
			}
		}
	}
	

	ai = {
		evaluation_chance = {
			value = 0.01 # AI shouldn't volunteer into power blocs all that often
		}		
	
		accept_score = {
			value = 0

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_BASE_ADDITIONAL"
				value = -10
			}

			if = {
				limit = {
					has_strategic_adjacency = scope:target_country
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_NEIGHBORS"
					value = 10
				}
			}

			if = {
				limit = {
					OR = {
						is_country_type = recognized
						is_country_type = colonial
					}
					scope:target_country = {
						is_country_type = unrecognized
					}
				}
				add = {
					desc = "concept_unrecognized_power_short"
					value = -200
				}
			}

			if = {
				limit = {
					has_strategy = ai_strategy_armed_isolationism
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ISOLATIONIST"
					add = -40
				}
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ARMY_STRENGTH"
				value = "scope:target_country.ai_army_comparison(root)"
				add = 0.25
				multiply = 25
				min = -200
				max = 50
			}

			add = {
				desc = "DIPLOMATIC_ACCEPTANCE_ATTITUDE"
				if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:target_country
								attitude = antagonistic
							}
							has_attitude = {
								who = scope:target_country
								attitude = domineering
							}
							has_attitude = {
								who = scope:target_country
								attitude = belligerent
							}
							has_attitude = {
								who = scope:target_country
								attitude = rebellious
							}
						}
					}
					value = -1000
				}
				else_if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:target_country
								attitude = wary
							}
							has_attitude = {
								who = scope:target_country
								attitude = disinterested
							}
						}
					}
					value = -10
				}
				else_if = {
					limit = {
						has_attitude = {
							who = scope:target_country
							attitude = cautious
						}
					}
					value = -50
				}				
				else_if = {
					limit = {
						OR = {
							has_attitude = {
								who = scope:target_country
								attitude = protective
							}
							has_attitude = {
								who = scope:target_country
								attitude = genial
							}
							has_attitude = {
								who = scope:target_country
								attitude = cooperative
							}
							has_attitude = {
								who = scope:target_country
								attitude = conciliatory
							}
						}
					}
					value = 10
				}
			}

			if = {
				limit = {
					"num_alliances_and_defensive_pacts_with_rivals(scope:target_country)" > 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ALLIED_TO_RIVALS"
					value = "num_alliances_and_defensive_pacts_with_rivals(scope:target_country)"
					multiply = -25
				}
			}

			if = {
				limit = {
					"num_alliances_and_defensive_pacts_with_allies(scope:target_country)" > 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_ALLIED_TO_ALLIES"
					value = "num_alliances_and_defensive_pacts_with_allies(scope:target_country)"
					multiply = 10
				}
			}

			if = {
				limit = {
					"num_shared_rivals(scope:target_country)" > 0
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_SHARED_RIVALS"
					value = "num_shared_rivals(scope:target_country)"
					multiply = 10
					max = 30
				}
			}

			if = {
				limit = {
					NOT = {
						any_scope_treaty = {
							binds = scope:target_country
							any_scope_article = {
								has_type = alliance
							}
						}
					}
					any_scope_treaty = {
						count > 0
						any_scope_article = {
							has_type = alliance
						}
					}
				}
				add = {
					desc = "DIPLOMATIC_ACCEPTANCE_NUM_ALLIES"
					value = {
						every_scope_treaty = {
							limit = {
								any_scope_article = {
									has_type = alliance
						 		}
							}
							add = 1
						}
					}
					multiply = -20
				}
			}

			if = {
				limit = {
					exists = power_bloc
					exists = scope:target_country.power_bloc
				}
				if = {
					limit = {
						power_bloc = scope:target_country.power_bloc
					}
					add = {
						desc = "DIPLOMATIC_ACCEPTANCE_SAME_POWER_BLOC"
						value = 15
					}
				}
				else_if = {
					limit = {
						is_power_bloc_leader = no
						scope:target_country = { is_power_bloc_leader = no }							
					}				
					add = {
						desc = "DIPLOMATIC_ACCEPTANCE_DIFFERENT_POWER_BLOC"
						value = -25
					}
				}
			}

			if = {
				limit = {
					NOT = { exists = scope:is_treaty_active }
					exists = scope:treaty
					scope:treaty.binding_period >= 1825
				}
				subtract = {
					desc = "DIPLOMATIC_ACCEPTANCE_BINDING_PERIOD"
					value = scope:treaty.binding_period
					divide = 3650
					subtract = 25
					max = 50
				}
			}
		}
		
		will_propose = {  
			would_accept_diplomatic_action = {
				actor = scope:target_country
				type = ut_influence_new_dawn_major_power
			}
		}
		
		propose_score = {
			value = 1
			#will do if can, but low chance
		}
	}
}

#copied force become subject
#Communist-to-Communist subjugation
ut_force_become_subject = {
	groups= {
		power_bloc_leader
	}
	show_in_lens = no

	texture = "gfx/interface/icons/diplomatic_action_icons/force_become_subject.dds"

	selectable = {
		is_power_bloc_leader = yes
		has_modifier = ut_new_dawn_core_communist_mod
	}

	potential = {
		power_bloc ?= {
			scope:target_country.power_bloc ?= this
		}
		OR = {
			has_law = law_type:law_cooperative_ownership
			has_law = law_type:law_command_economy
		}
	}

	possible = {
		custom_tooltip = ut_has_new_dawn_progress_50
		ut_has_new_dawn_progress_communist = {
			VALUE = 100
		}
		scope:target_country = {
			is_subject = no
			country_rank < rank_value:unrecognized_major_power
			is_diplomatic_play_committed_participant = no
			OR = {
				has_law = law_type:law_cooperative_ownership
				has_law = law_type:law_command_economy
				has_modifier = ut_new_dawn_core_communist_mod
			}
		}

		power_bloc = {
			current_cohesion_percentage >= 0.5
		}
		scope:target_country = {
			tenure_in_current_power_bloc_years >= 5
		}
		prestige >= {
			value = scope:target_country.prestige
			multiply = 5
		}
	}

	accept_effect = {
		if = {
			limit = { is_country_type = unrecognized }
			create_diplomatic_pact = {
				country = scope:target_country
				type = tributary
			}
		}
		else = {
			create_diplomatic_pact = {
				country = scope:target_country
				type = protectorate
			}
		}
		create_unidirectional_truce = { country = scope:target_country months = 60 }
		hidden_effect = { scope:target_country = { add_liberty_desire = 30 } } # Starts at a higher LD level

		power_bloc = {
			add_cohesion_percent = {
				value = scaled_cohesion_cost_subjugation
			}
		}

		custom_tooltip = ut_add_new_dawn_progress_neg_50
		ut_add_new_dawn_progress_communist = {
			VALUE = -50
		}
	}

	ai = {
		evaluation_chance = {
			value = 0.025 # AI shouldn't use this very often
		}

		will_propose = {
			power_bloc ?= { current_cohesion_percentage >= 0.6 }
			OR = {
				has_attitude = {
					who = scope:target_country
					attitude = domineering
				}
				power_bloc ?= { current_cohesion_percentage >= 0.8 }
			}
			OR = {
				AND = {
					has_attitude = {
						who = scope:target_country
						attitude = domineering
					}
					power_bloc ?= { current_cohesion_percentage >= 0.8 }
				}
				scope:target_country = {
					is_country_type = unrecognized
				}
			}
		}

		propose_score = {
			value = 10
		}
	}
}

#copied force regime change
#Fascist force Extraction + Corporate State
ut_force_regime_change = {
	groups= {
		power_bloc_leader
	}
	show_in_lens = no

	texture = "gfx/interface/icons/diplomatic_action_icons/force_regime_change.dds"

	selectable = {
		is_power_bloc_leader = yes
		has_modifier = ut_new_dawn_core_communist_mod
	}

	potential = {
		power_bloc ?= {
			scope:target_country.power_bloc ?= this
		}
		OR = {
			has_subject_relation_with = scope:target_country
			country_rank <= rank_value:minor_power
		}
		custom_tooltip = ut_has_new_dawn_progress_50
		ut_has_new_dawn_progress_fascist = {
			VALUE = 100
		}
	}

	possible = {
		custom_tooltip = {
			text = "FORCE_REGIME_CHANGE_GOV_PRINCIPLES_DIST_POWER_PROGRESSIVENESS_DIFFERENCE"
			OR = {
				50 <= {
					add = {
						add = active_law:lawgroup_governance_principles.type.progressiveness
						subtract = scope:target_country.active_law:lawgroup_governance_principles.type.progressiveness
						abs = yes
					}
				}
				50 <= {
					add = {
						add = active_law:lawgroup_distribution_of_power.type.progressiveness
						subtract = scope:target_country.active_law:lawgroup_distribution_of_power.type.progressiveness
						abs = yes
					}
				}
			}
		}

		power_bloc = {
			current_cohesion_percentage >= 0.1
		}
		scope:target_country = {
			tenure_in_current_power_bloc_years >= 5
		}
		prestige >= {
			value = scope:target_country.prestige
			multiply = 3
		}
	}

	accept_effect = {
		scope:target_country = {
			activate_law = law_type:law_corporate_state
			activate_law = law_type:law_extraction_economy
			activate_law = law_type:law_subjecthood
		}

		power_bloc = {
			add_cohesion_percent = scaled_cohesion_cost
		}

		custom_tooltip = ut_add_new_dawn_progress_neg_50
		ut_add_new_dawn_progress_fascist = {
			VALUE = -50
		}
	}

	ai = {
		evaluation_chance = {
			value = 0.025 # AI shouldn't use this very often
		}

		will_propose = {
			power_bloc = { current_cohesion_percentage >= 0.75 }
