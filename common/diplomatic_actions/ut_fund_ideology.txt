#copied from fund lobbies
ut_fund_ideology = {
	groups= {
		general
	}
	requires_approval = no
	should_notify_third_parties = no

	unlocking_technologies = {
		union_new_mandates
	}

	selectable = {
		NOT = { is_country_type = decentralized }
		is_revolutionary = no
	}

	potential = {
		scope:target_country = {
			NOT = { is_country_type = decentralized }
			is_revolutionary = no
			has_journal_entry = je_ut_new_dawn
		}
	}

	pact = {
		cost = 100

		is_two_sided_pact = no

		second_modifier = {
			country_treaty_leverage_generation_mult = 0.1
		}

		second_foreign_pro_country_lobby_member_modifier = {
			interest_group_pop_attraction_mult = 0.1
		}

		second_foreign_anti_country_lobby_member_modifier = {
			interest_group_pop_attraction_mult = -0.1
		}

		income_transfer = 0.20
		max_paying_country_income_to_transfer = 0.10
		second_country_gets_income_transfer = yes
		income_transfer_based_on_second_country = yes
		income_transfer_to_pops = {
			allow_non_fully_accepted = no
			upper_strata_pops = {
				value = 0.5
				if = {
					limit = {
						OR = {
							has_law = law_type:law_autocracy
							has_law = law_type:law_oligarchy
						}
					}
					add = 0.5
				}
				else_if = {
					limit = {
						OR = {
							has_law = law_type:law_wealth_voting
							has_law = law_type:law_landed_voting
						}
					}
					add = 0.25
				}
				else_if = {
					limit = {
						has_law = law_type:law_anarchy
					}
					subtract = 0.25
				}

				max = 1
				min = 0
			}
			middle_strata_pops = {
				value = 0.25

				if = {
					limit = {
						OR = {
							has_law = law_type:law_census_voting
							has_law = law_type:law_universal_suffrage
						}
					}
					add = 0.25
				}

				max = 1
				min = 0
			}
			lower_strata_pops = {
				value = 0.0

				if = {
					limit = {
						OR = {
							has_law = law_type:law_universal_suffrage
							has_law = law_type:law_anarchy
						}
					}
					add = 0.25
				}

				max = 1
				min = 0
			}
		}

		monthly_effect = {
			custom_tooltip = ut_fund_ideology_effect
			if = {
				limit = {
					has_modifier = ut_new_dawn_core_autocratic_mod
				}
				scope:target_country = {
					je:je_ut_new_dawn = {
						add_progress = {
							name = ut_new_dawn_autocrat_bar
							value = 0.25
						}
					}
				}
			}
			else_if = {
				limit = {
					has_modifier = ut_new_dawn_core_democratic_mod
				}
				scope:target_country = {
					je:je_ut_new_dawn = {
						add_progress = {
							name = ut_new_dawn_democrat_bar
							value = 0.25
						}
					}
				}
			}
			else_if = {
				limit = {
					has_modifier = ut_new_dawn_core_fascist_mod
				}
				scope:target_country = {
					je:je_ut_new_dawn = {
						add_progress = {
							name = ut_new_dawn_fascist_bar
							value = 0.25
						}
					}
				}
			}
			else_if = {
				limit = {
					has_modifier = ut_new_dawn_core_communist_mod
				}
				scope:target_country = {
					je:je_ut_new_dawn = {
						add_progress = {
							name = ut_new_dawn_communist_bar
							value = 0.25
						}
					}
				}
			}
			else = {
				custom_tooltip = error
			}
		}

		requirement_to_maintain = {
			trigger = { 
				scope:target_country = {
					NOT = { ut_new_dawn_has_core_trigger = yes }
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				has_diplomatic_relevance = scope:target_country
			}
		}

	 	requirement_to_maintain = {
			trigger = {
				NOT = {
					scope:target_country = {
						has_diplomatic_pact = {
							who = root
							type = rivalry
						}
					}
				}
			}
		}

	 	requirement_to_maintain = {
			trigger = {
				NOT = {
					has_diplomatic_pact = {
						who = scope:target_country
						type = rivalry
					}
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_NOT_DECENTRALIZED"
					NOT = { is_country_type = decentralized }
					scope:target_country = {
						NOT = { is_country_type = decentralized }
					}
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "DIPLOMATIC_PACT_REQUIREMENT_AT_PEACE"
					NOT = { has_war_with = scope:target_country }
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				custom_tooltip = {
					text = "EQUAL_OR_BETTER_RANK_THAN_TARGET"
					country_rank >= scope:target_country.country_rank
				}
			}
		}

		requirement_to_maintain = {
			trigger = {
				in_default = no
			}
		}
	}

	ai = {
		max_influence_spending_fraction = 0.20

		evaluation_chance = {
			value = 0.1

			add = {
				value = country_rank
				multiply = 0.05
			}
		}

		propose_score = {
			value = 10

			if = {
				limit = {
					has_journal_entry = je_hawaiian_interest
					c:HAW ?= {
						this = scope:target_country
					}
				}
				add = 100
			}
		}

		will_propose = {
			#todo
			always = no
			scaled_debt <= 0
			is_at_war = no
			net_fixed_income > 0
			has_attitude = {
				who = scope:target_country
				attitude = protective
			}
			OR = {
				num_income_transfer_pacts < 1
				AND = {
					num_income_transfer_pacts < 2
					income >= 200000
				}
				AND = {
					num_income_transfer_pacts < 3
					income >= 400000
				}
			}

			OR = {
				root = {
					owes_obligation_to = scope:target_country
				}
				NOT = {
					scope:target_country = {
						owes_obligation_to = root
					}
				}
				AND = {
					is_at_war = no
					scope:target_country = { is_at_war = yes }
				}
			}
			# We want to have at least five times their income, and at least 50k income in general
			OR = {
				AND = {
					income >= 50000
					net_fixed_income >= 4000
					scope:target_country.fixed_income <= 10000
				}
				AND = {
					income >= 75000
					net_fixed_income >= 6000
					scope:target_country.fixed_income <= 15000
				}
				AND = {
					income >= 100000
					net_fixed_income >= 8000
					scope:target_country.fixed_income <= 20000
				}
				AND = {
					income >= 125000
					net_fixed_income >= 10000
					scope:target_country.fixed_income <= 25000
				}
				AND = {
					income >= 150000
					net_fixed_income >= 12000
					scope:target_country.fixed_income <= 30000
				}
				AND = {
					income >= 175000
					net_fixed_income >= 14000
					scope:target_country.fixed_income <= 35000
				}
				AND = {
					income >= 200000
					net_fixed_income >= 16000
					scope:target_country.fixed_income <= 40000
				}
				AND = {
					income >= 225000
					net_fixed_income >= 18000
					scope:target_country.fixed_income <= 45000
				}
				AND = {
					income >= 250000
					net_fixed_income >= 20000
					scope:target_country.fixed_income <= 50000
				}
			}
		}

		will_break = {
			OR = {
				scaled_debt > 0.25
				AND = {
					scaled_debt > 0
					OR = {
						net_fixed_income < 0
						is_at_war = yes
					}
				}
				NOT = {
					has_attitude = {
						who = scope:target_country
						attitude = protective
					}
				}
				num_income_transfer_pacts > 3
				AND = {
					num_income_transfer_pacts > 1
					OR = {
						net_fixed_income < -10000
						income < 300000
					}
				}
				AND = {
					num_income_transfer_pacts > 2
					OR = {
						net_fixed_income < -5000
						income < 500000
					}
				}
				NOR = {
					root = {
						owes_obligation_to = scope:target_country
					}
					NOT = {
						scope:target_country = {
							owes_obligation_to = root
						}
					}
					AND = {
						is_at_war = no
						scope:target_country = { is_at_war = yes }
					}
				}
				# Threshold for breaking off a bankroll is higher than for starting it
				# We also don't look at net_fixed_income here since starting a bankroll changes it
				NOR = {
					AND = {
						income >= 40000
						scope:target_country.fixed_income <= 12500
					}
					AND = {
						income >= 50000
						scope:target_country.fixed_income <= 15000
					}
					AND = {
						income >= 75000
						scope:target_country.fixed_income <= 20000
					}
					AND = {
						income >= 100000
						scope:target_country.fixed_income <= 25000
					}
					AND = {
						income >= 125000
						scope:target_country.fixed_income <= 30000
					}
					AND = {
						income >= 150000
						scope:target_country.fixed_income <= 35000
					}
					AND = {
						income >= 175000
						scope:target_country.fixed_income <= 40000
					}
					AND = {
						income >= 200000
						scope:target_country.fixed_income <= 45000
					}
					AND = {
						income >= 225000
						scope:target_country.fixed_income <= 50000
					}
					AND = {
						income >= 250000
						scope:target_country.fixed_income <= 55000
					}
				}
			}
		}
	}
}
