je_ut_software_era_6 = {
	icon = "gfx/interface/icons/goods_icons/software.dds"

	group = je_group_technology
	
	modifiers_while_active = {
	}
	
	is_shown_when_inactive = {
		always = yes
	}

	possible = {
		has_technology_researched = electronics
	}

	immediate = {
		set_variable = {
			name = je_ut_software_points_bar_progress_buildings_era_6
			value = 0
		}
	}
	
	scripted_button = je_ut_software_era_6_1_assembler_language
	scripted_button = je_ut_software_era_6_2_assembler_language
	scripted_button = je_ut_software_era_6_3_high_level_language

	scripted_progress_bar = je_ut_software_points_bar_era_6

	complete = {
		has_technology_researched = computers
		has_modifier = je_ut_software_era_6_3_high_level_language_mod
	}

	on_complete = {
		set_variable = je_ut_software_era_6_done
		add_journal_entry = {
			type = je_ut_software_era_7
		}
		remove_variable = je_ut_software_points_bar_progress_buildings_era_6

		if = {
			limit = {
				is_ai = yes
				capital = {
					NOT = { has_building = building_software_company }
				}
			}
			capital = {
				create_building = {
					building = building_software_company
					level = 1
					reserves=1
					subsidized = yes
				}
			}
		}
	}

	status_desc = {
	}

	on_monthly_pulse = {
		effect = {
			
			set_variable = {
				name = je_ut_software_points_bar_progress_buildings_era_6
				value = 0
			}

			if = {
				limit = {
					is_player = yes
				}
				#for player
				every_scope_building = {
					limit = {
						is_building_type = building_software_company
						level >= 1
					}
					ROOT = {
						set_local_variable = {
							name = je_ut_software_points_bar_progress_buildings_loc
							value = prev.occupancy
						}
						change_local_variable = {
							name = je_ut_software_points_bar_progress_buildings_loc
							multiply = prev.level
						}
						change_variable = {
							name = je_ut_software_points_bar_progress_buildings_era_6
							add = local_var:je_ut_software_points_bar_progress_buildings_loc
						}
					}
					#cap at 10
					if = {
						limit = {
							has_variable = je_ut_software_points_bar_progress_buildings_era_6
							var:je_ut_software_points_bar_progress_buildings_era_6 > 10
						}
						set_variable = {
							name = je_ut_software_points_bar_progress_buildings_era_6
							value = 10
						}
					}
				}
			}
			else = {
				#for ai flat calculations
				set_variable = {
					name = je_ut_software_points_bar_progress_buildings_era_6
					value = 0.5
				}
				if = {
					limit = {
						country_rank >= rank_value:major_power
					}
					change_variable = {
						name = je_ut_software_points_bar_progress_buildings_era_6
						add = 1
					}
				}
				if = {
					limit = {
						has_building = building_software_company
					}
					change_variable = {
						name = je_ut_software_points_bar_progress_buildings_era_6
						add = 2
					}
				}
			}

		}
	}

	weight = 100

	should_be_pinned_by_default_uninvolved_or_context = yes
}
