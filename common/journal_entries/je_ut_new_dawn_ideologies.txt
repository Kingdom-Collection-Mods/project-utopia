ut_je_propaganda_source = {
	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	group = je_group_utopia

	immediate = {
		set_variable = {
			name = ut_propaganda_source_var
			value = 0
		}
	}

	modifiers_while_active = {
		ut_je_propaganda_source_mod
	}

	scripted_button = ut_new_dawn_propaganda_source_abort

	on_monthly_pulse = {
		effect = {
			change_variable = {
				name = ut_propaganda_source_var
				add = 1
			}
			if = {
				limit = {
					has_modifier = ut_new_dawn_core_external_offensive_focus_mod
				}
				change_variable = {
					name = ut_propaganda_source_var
					add = 0.2
				}
			}
		}
	}

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
	}

	on_complete = {
		hidden_effect = {
			remove_variable = ut_new_dawn_propaganda_source_defend_once_var
			set_variable = {
				name = ut_je_propaganda_source_target_var
				value = scope:target
			}
			trigger_event = {
				id = event_new_dawn.25
				popup = yes
			}
		}
		scope:target = {
			custom_tooltip = ut_add_new_dawn_progress_neg_25
			ut_add_new_dawn_core_progress = {
				VALUE = -25
			}
			
			set_variable = {
				name = ut_je_propaganda_target_source_var
				value = root
			}
			trigger_event = {
				id = event_new_dawn.24
				popup = yes
			}
		}
		if = {
			limit = {
				has_modifier = ut_new_dawn_core_communist_mod
			}
			scope:target = {
				add_modifier = {
					name = ut_propaganda_communist_mod
					days = 730 #2 years
					#is_decaying = yes
				}
			}
		}
		else_if = {
			limit = {
				has_modifier = ut_new_dawn_core_democratic_mod
			}
			scope:target = {
				add_modifier = {
					name = ut_propaganda_democratic_mod
					days = 730 #2 years
					#is_decaying = yes
				}
			}
		}
		else_if = {
			limit = {
				has_modifier = ut_new_dawn_core_fascist_mod
			}
			scope:target = {
				add_modifier = {
					name = ut_propaganda_fascist_mod
					days = 730 #2 years
					#is_decaying = yes
				}
			}
		}
		else_if = {
			limit = {
				has_modifier = ut_new_dawn_core_autocratic_mod
			}
			scope:target = {
				add_modifier = {
					name = ut_propaganda_autocratic_mod
					days = 730 #2 years
					#is_decaying = yes
				}
			}
		}
	}

	fail = {
		custom_tooltip = {
			text = ut_je_propaganda_source_abort_tt
			has_variable = ut_propaganda_source_abandon_var
		}
	}

	on_fail = {
		remove_variable = ut_propaganda_source_abandon_var
		remove_variable = ut_propaganda_source_var
		remove_variable = ut_new_dawn_propaganda_source_defend_once_var
		scope:target = {
			set_variable = ut_propaganda_target_abandon_var
		}
	}

	current_value = {
		value = root.var:ut_propaganda_source_var
	}

	goal_add_value = {
		value = 12 # 1 year
	}

	progressbar = yes

	weight = 10

	should_be_pinned_by_default_uninvolved_or_context = yes
}

ut_je_ally_influence_petition = {
	icon = "gfx/interface/icons/new_dawn_icons/ut_new_dawn.dds"

	group = je_group_utopia

    scripted_button = je_government_petition_go_away_button
    scripted_button = je_government_petition_extension_button

    modifiers_while_active = {
    }

    immediate = {
		scope:target = {
			save_scope_as = convert_target_scope
		}
    }

    status_desc = {
		first_valid = {
			triggered_desc = {
				desc = ut_je_ally_influence_petition_communist
				trigger = {
					has_modifier = ut_new_dawn_core_communist_mod
				}
			}
			
			triggered_desc = {
				desc = ut_je_ally_influence_petition_democratic
				trigger = {
					has_modifier = ut_new_dawn_core_democratic_mod
				}
			}
			
			triggered_desc = {
				desc = utopia.error
			}
		}
	}

    complete = {
		trigger_if = {
			limit = {
				has_modifier = ut_new_dawn_core_democratic_mod
			}
			scope:target = {
				OR = {
					has_law_or_variant = law_type:law_universal_suffrage
					any_scope_treaty = {
						any_scope_article = {
							has_type = ut_democracy_article
						}
					}
				}
				trigger_if = {
					limit = {
						country_rank >= rank_value:great_power
					}
					has_modifier = ut_new_dawn_core_democratic_mod
				}
			}
		}
		trigger_if = {
			limit = {
				has_modifier = ut_new_dawn_core_communist_mod
			}
			scope:target = {
				OR = {
					has_law_or_variant = law_type:law_council_republic
					any_scope_treaty = {
						any_scope_article = {
							has_type = ut_communism_article
						}
					}
				}
				trigger_if = {
					limit = {
						country_rank >= rank_value:great_power
					}
					has_modifier = ut_new_dawn_core_communist_mod
				}
			}
		}
    }

    on_complete = {
		custom_tooltip = ut_add_new_dawn_progress_5
		ut_add_new_dawn_core_progress = { 
			VALUE = 5
		}
		add_modifier = {
			name = modifier_successfully_met_petition_legitimacy
			days = long_modifier_time
			multiplier = {
				value = 0.5
				if = {
					limit = {
						scope:journal_entry ?= {
							has_variable = asked_for_extension_var
						}
					}
					divide = 2
				}
			}
			is_decaying = yes
		}
    }

    invalid = {
		trigger_if = {
			limit = {
				has_modifier = ut_new_dawn_core_democratic_mod
			}
			NOT = { has_modifier = ut_new_dawn_core_democratic_mod }
		}
		trigger_else_if = {
			limit = {
				has_modifier = ut_new_dawn_core_communist_mod
			}
			NOT = { has_modifier = ut_new_dawn_core_communist_mod }
		}
		trigger_else = {
			always = yes
		}
    }

    fail = {
		exists = scope:convert_target_scope
		custom_tooltip = {
			text = has_been_abandoned_tt
			scope:journal_entry ?= {
				has_variable = has_been_abandoned
			}
		}
    }

    on_timeout = {
		custom_tooltip = ut_add_new_dawn_progress_neg_5
		ut_add_new_dawn_core_progress = { 
			VALUE = -5
		}
    }

    on_fail = {
		custom_tooltip = ut_add_new_dawn_progress_neg_5
		ut_add_new_dawn_core_progress = { 
			VALUE = -5
		}
    }

    on_invalid = {
    }

    timeout = {
        value = 1825 # 5 years
        if = {
            limit = {
                scope:journal_entry = {
                    has_variable = asked_for_extension_var
                }
            }
            add = 1095
        }
    }

    weight = 1000

    	
	should_be_pinned_by_default_uninvolved_or_context = yes
}
