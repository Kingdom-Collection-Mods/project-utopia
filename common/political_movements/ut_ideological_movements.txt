ut_movement_cosmopolitan = {
	category = movement_category_ideological

	ideology = ut_ideology_cosmopolitan	
	character_ideologies = {
		ut_ideology_cosmopolitan_leader
	}
	
	disband_trigger = {
		has_law = law_type:law_multicultural_globalism
		OR = {
			has_law = law_type:law_free_trade_union
			has_law = law_type:law_protectionism_union
		}
	}
	on_disbanded = {}
	
	creation_trigger = {
		has_technology_researched = anti_discrimination_laws
	}
	creation_weight = {
		value = 1000
	}	
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	
	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_humanitarian
			has_ideology = ideology:ideology_humanitarian_royalist
			has_ideology = ideology:ut_social_feminism_leader
			OR = {
				is_interest_group_type = ig_intelligentsia
				is_interest_group_type = ig_trade_unions
				is_interest_group_type = ig_urbanists
				is_interest_group_type = ig_industrialists
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_humanitarian
					has_ideology = ideology:ideology_humanitarian_royalist
					has_ideology = ideology:ut_social_feminism_leader
				}
			}
			multiply = {
				value = 5
			}
		}
	}
	
	
	pop_support_trigger = {
		NOR = {
			is_pop_type = clergymen
			is_pop_type = peasants
			is_pop_type = slaves
			is_pop_type = farmers
		}
	}

	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_academics
		movement_support_service_workers
		movement_support_capitalists
		movement_support_engineers
		movement_support_clerks
		movement_support_soldiers
	}
	
	pop_support_weight = {	
		add = {
			desc = "URBAN_STATE"
			value = 20
			multiply = state.state_urbanization_rate
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 15
					desc = "POP_CAPITALISTS"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 50
					desc = "POP_ACADEMICS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = service_workers
				}
				add = {
					value = 25
					desc = "POP_SERVICE_WORKERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 25
					desc = "POP_BUREAUCRATS"
				}
			}
			else = {
				add = {
					value = 10
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 9
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 6
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}
	}
	
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		
		weight = {
			value = 100
		}	
	
		state_weight = {
			value = 1		
		}
	}
	
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		
		weight = {
			value = 1000
		}		
	}
}

#based on fascists
ut_movement_racists = {
	category = movement_category_ideological

	# Fascists care about a lot of law groups, so the impact of each should be lesser
	active_law_radicalism_multiplier = 0.5

	ideology = ideology_fascist_movement
	character_ideologies = {
		ideology_fascist
		ideology_integralist
		ideology_corporatist_leader
		ut_ideology_autocratic_leader
		ut_ideology_national_socialist_leader
	}

	creation_trigger = {
		has_technology_researched = national_socialism
	}
	creation_weight = {
		value = 100
		if = {
			limit = {
				owner ?= { # If a movement of this type exists in a neighbouring country, make it more likely to spawn here as well
					any_neighbouring_state = {
						owner = {
							prestige > ROOT.owner.prestige
							any_political_movement = {
								is_political_movement_type = movement_fascist
							}
						}
					}
				}
			}
			multiply = {
				value = 2
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
		owner ?= {
			if = {
				limit = {
					any_primary_culture = {
						has_discrimination_trait = language_lusophone
					}
				}
				prev = {
					set_core_ideology = ideology_integralist_movement
				}
			}
		}
	}

	character_support_trigger = {
		OR = {
			has_ideology = ideology:ideology_fascist
			has_ideology = ideology:ideology_integralist
			has_ideology = ideology:ideology_corporatist_leader
			has_ideology = ideology:ideology_ethno_nationalist
			has_ideology = ideology:ut_ideology_autocratic_leader
			has_ideology = ideology:ut_ideology_national_socialist_leader
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_fascist
					has_ideology = ideology:ideology_integralist
				}
			}
			multiply = {
				value = 6 #5
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_corporatist_leader
					has_ideology = ideology:ideology_ethno_nationalist
				}
			}
			multiply = {
				value = 3 #2
			}
		}
		else_if = {
			limit = {
				OR = {
					has_ideology = ideology:ut_ideology_autocratic_leader
					has_ideology = ideology:ut_ideology_national_socialist_leader
				}
			}
			multiply = {
				value = 4
			}
		}
	}

	pop_support_trigger = {
		NOT = {
			is_pop_type = slaves
		}
	}

	pop_support_factors = {
		movement_support_high_literacy
		movement_support_middle_below_expected_sol
		movement_support_radicalization
		movement_support_unemployed
		movement_support_capitalists
		movement_support_aristocrats
		movement_support_shopkeepers
		movement_support_officers
		movement_support_farmers
		movement_support_clerks
		movement_support_soldiers
		movement_support_number_of_claims
	}

	pop_support_weight = {
		if = {
			limit = {
				exists = scope:political_movement
				scope:political_movement ?= {
						OR = {
							is_being_suppressed = yes
							is_being_bolstered = yes
						}
					}
				}
			if = {
				limit = {
					scope:political_movement = {
						is_being_suppressed = yes
					}
				}
				add = {
					value = suppression_base_value
					multiply = {
						value = -0.5
						add = owner.modifier:country_suppression_attraction_factor
					}
					desc = "concept_suppressed"
				}
			}
			else = {
				add = {
					value = bolstering_base_value
					multiply = {
						value = 0.5
						add = owner.modifier:country_bolster_attraction_factor
					}
					desc = "concept_bolstered"
				}
			}
		}

		add = {
			value = 3
			desc = "POP_BASE_SUPPORT"
		}

		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 7
					desc = "POP_CAPITALISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = aristocrats
				}
				add = {
					value = 2
					desc = "POP_ARISTOCRATS" # Fascism is too populist
				}
			}
			else = {
				add = {
					value = 4
					desc = "UPPER_NO_ICON"
				}
			}

			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_anarchist
								is_political_movement_type = movement_communist
							}
							political_movement_support >= 0.10 # (chuckles) im in danger
						}
					}
				}
				multiply = {
					value = 1.5
					desc = "STRONG_SOCIALIST_MOVEMENT"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				add = {
					value = 15
					desc = "POP_SHOPKEEPERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = officers
				}
				add = {
					value = 10
					desc = "POP_OFFICERS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = farmers
				}
				add = {
					value = 10
					desc = "POP_FARMERS"
				}
			}
			else = {
				add = {
					value = 7
					desc = "MIDDLE_NO_ICON"
				}
			}

			if = {
				limit = {
					owner ?= {
						any_political_movement = {
							OR = {
								is_political_movement_type = movement_socialist
								is_political_movement_type = movement_anarchist
								is_political_movement_type = movement_communist
							}
							political_movement_support >= 0.10
						}
					}
				}
				multiply = {
					value = 1.25
					desc = "STRONG_SOCIALIST_MOVEMENT"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 9
					desc = "POP_CLERKS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 6
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 3
					desc = "LOWER_NO_ICON"
				}
			}
		}

		owner = {
			if = {
				limit = {
					number_of_claims > 0
				}

				add = {
					value = 5 # Add a base value for having claims at all
					add = {
						value = number_of_claims
						multiply = 5
					}
					max = 25
					desc = "NUMBER_OF_CLAIMS"
				}
			}
		}

		if = {
			limit = {
				standard_of_living < state.average_expected_sol
				strata = middle
			}
			add = { # Increases likelihood of support if impoverished
				desc = "IS_BELOW_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				max = 10
			}
		}
		else_if = {
			limit = {
				standard_of_living > state.average_expected_sol
				strata = middle
			}
			add = { # Decreases likelihood of support if well-off
				desc = "IS_ABOVE_EXPECTED_SOL"
				value = state.average_expected_sol
				subtract = standard_of_living
				min = -5
			}
		}

		if = {
			limit = {
				is_employed = no
			}
			multiply = {
				value = 1.5
				desc = "UNEMPLOYED"
			}
		}

		multiply = { # Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.10
		}

		multiply = {
			value = 0.012
			multiply = pop_acceptance
			desc = "ACCEPTANCE"
		}
	}

	revolution = {
		possible = {
			political_movement_support >= 0.05
		}

		weight = {
			value = 100
		}

		state_weight = {
			value = 1
		}
	}

	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}

		weight = {
			value = 1000
		}
	}

	additional_radicalism_factors = {
		owner = {
			if = {
				limit = {
					owner.number_of_claims > 0
				}

				add = {
					value = 5 # Add a base value for having claims at all
					add = owner.number_of_claims
					multiply = 0.025
					max = 0.3
					desc = "NUMBER_OF_CLAIMS"
