# Project Utopia - Late Game & Futuristic Formable Nations
# These formables become available in the later eras (1936+, Eras 6-8)
# and leverage the new technologies, goods, and mechanics added by Utopia.
# Requires union_new_mandates / union_new_mandates_two (your mod's international
# cooperation techs) or vanilla pan-nationalism as prerequisites.

### EUROPEAN FEDERATION
EUR = {
	use_culture_states = yes

	is_major_formation = yes

	unification_play = dp_unify_european_federation
	leadership_play = dp_leadership_european_federation

	required_states_fraction = 0.75

	ai_will_do = {
		has_technology_researched = union_new_mandates
	}

	potential = {
		has_technology_researched = union_new_mandates
		has_dlc_feature = rp1_content
	}

	possible = {
		has_technology_researched = union_new_mandates
		is_in_geographic_region = geographic_region_europe
		capital = { is_in_geographic_region = geographic_region_europe }
		NOT = {
			is_subject = yes
		}
		country_rank >= rank_value:major_power
	}

	max_num_formation_candidates = 5
	can_be_formation_candidate = {
		country_rank >= rank_value:major_power
		is_in_geographic_region = geographic_region_europe
		has_technology_researched = union_new_mandates
	}

	can_be_unification_target = {
		is_in_geographic_region = geographic_region_europe
		has_technology_researched = union_new_mandates
	}

	should_target_state_in_unification_play = {}
}

### ARCTIC COUNCIL
ARC = {
	use_culture_states = yes

	required_states_fraction = 0.5

	ai_will_do = {
		always = yes
	}

	potential = {
		has_technology_researched = union_new_mandates
	}

	possible = {
		has_technology_researched = union_new_mandates
		OR = {
			country_has_primary_culture = cu:swedish
			country_has_primary_culture = cu:norwegian
			country_has_primary_culture = cu:danish
			country_has_primary_culture = cu:icelandic
			country_has_primary_culture = cu:russian
			country_has_primary_culture = cu:canadian
			c:USA ?= this
		}
		NOT = {
			is_subject = yes
		}
		country_rank >= rank_value:major_power
	}
}
