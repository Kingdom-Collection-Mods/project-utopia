ut_new_dawn_core_external_offensive_focus_add = {
	name = "ut_new_dawn_core_external_offensive_focus_add"
	desc = "ut_new_dawn_core_external_offensive_focus_desc"

	visible = { 
		ut_new_dawn_has_core_trigger = yes
		is_player = yes
		NOT = { has_modifier = ut_new_dawn_core_external_offensive_focus_mod }	
	}

	possible = {
		custom_tooltip = {
			text = ut_has_new_dawn_progress_50
			ut_has_new_dawn_progress_core = {
				VALUE = 50
			}
		}
	}

	effect = {
		custom_tooltip = ut_add_new_dawn_progress_neg_1
		ut_add_new_dawn_core_progress = {
			VALUE = -1
		}
		add_modifier = ut_new_dawn_core_external_offensive_focus_mod
	}
	#ai never
	ai_chance = {
		value = 0
	}
}
