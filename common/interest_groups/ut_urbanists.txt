ig_urbanists = {
	color = hsv{ 0.55 0.4 0.9 }
	texture = "gfx/interface/icons/ig_icons/urbanists.dds"
	layer = "revolution_dynamic_intelligentsia"
	index = 3

	ideologies = {
		ideology_urbanist_power_base
		ideology_urbanist_economy_base
		ideology_urbanist_human_rights_base
	}
	
	character_ideologies = {
		ideology_moderate
		ut_ideology_environmentalist_leader
		ideology_authoritarian
		ideology_corporatist_leader
		ideology_liberal_leader
		ideology_ethno_nationalist
		ideology_fascist
		ideology_feminist
		ideology_humanitarian
		ideology_pacifist
		ideology_positivist
		ideology_reformer
		ideology_republican_leader
		ideology_social_democrat
		ideology_sovereignist_leader
	}

	enable = {
		ut_service_workers_possible = yes
	}

	# Deprecated; use on_enable effect to assign traits instead
	traits = {
		ig_trait_urbanist_unhappy
		ig_trait_urbanist_happy
		ig_trait_urbanist_loyal
	}

	on_enable = {
		ig:ig_urbanists = {
			set_ig_trait = ig_trait:ig_trait_urbanist_unhappy
			set_ig_trait = ig_trait:ig_trait_urbanist_happy
			set_ig_trait = ig_trait:ig_trait_urbanist_loyal
		}
		
		if = {
			limit = {
				has_technology_researched = feminism
			}
			every_interest_group = {
				limit = {
					has_ideology = ideology:ideology_liberal
				}
				remove_ideology = ideology_liberal
				add_ideology = ideology_liberal_modern
			}
		}
	}
	on_disable = {}
	priority_cultures = { 
	}
	on_character_ig_membership = {}

	pop_potential = {
		OR = {
			is_pop_type = service_workers
			AND = {
				literacy_rate > 0.75
				owner = {
					pop_type_percent_country = {
						pop_type = service_workers
						percent > 0.01
					}
				}
				OR = {
					is_pop_type = academics
					is_pop_type = bureaucrats
					is_pop_type = capitalists
					is_pop_type = clergymen
					is_pop_type = engineers
				}
			}
		}
	}

	pop_weight = {
		value = 0

		add = {
			desc = "POP_SERVICE_WORKERS"
			if = {
				limit = {
					is_pop_type = service_workers
				}
				value = 200
			}
		}

		add = {
			desc = "POP_ACADEMICS"
			if = {
				limit = {
					is_pop_type = academics
				}
				value = 100
			}
		}

		add = {
			desc = "POP_BUREAUCRATS"
			if = {
				limit = {
					is_pop_type = engineers
				}
				value = 50
			}
		}

		add = {
			desc = "POP_CLERKS"
			if = {
				limit = {
					is_pop_type = machinists
				}
				value = 25
			}
		}

		add = {
			desc = "POP_CLERKS"
			if = {
				limit = {
					is_pop_type = shopkeepers
				}
				value = 25
			}
		}
		
		add = {
			desc = "ADMINISTRATION_ROLE"
			if = {
				limit = {
					is_pop_type = capitalists
				}
				value = 25
			}
		}

		if = {
			limit = {
				literacy_rate >= 0.90
			}
			add = {
				desc = "POP_LITERACY"
				value = literacy_rate
				multiply = 10
			}
		}

		if = {
			limit = {
				standard_of_living > 25
			}
			add = {
				desc = "STANDARD_OF_LIVING"
				value = standard_of_living
				subtract = 25
			}
		}

		
		add = {
			desc = "URBAN_POP"
			if = {
				limit = {
					NOR = {
						pop_employment_building_group = bg_agriculture
						pop_employment_building_group = bg_ranching
						pop_employment_building_group = bg_plantations
					}
				}
				value = 25
				add = this.standard_of_living
			}
		}

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader ?= {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}

		multiply = {
			desc = "decentralized_power"
			value = 1.0

			if = {
				limit = {
					owner = {
						is_country_type = decentralized
					}
				}
				value = 0.0
			}
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_LOYALISTS"
				value = pop_loyalist_fraction
				divide = 4
				add = 1
			}
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				divide = 4
				add = 1
			}
		}
	}

	monarch_weight = {
		value = 1
		# Monarch is not likely to adopt a marginal IG
		if = {
			limit = {
				is_marginal = yes
			}
			multiply = {
				value = 0.1
			}
		}
		# Monarch is more likely to adopt a Powerful IG
		if = {
			limit = {
				is_powerful = yes
			}
			multiply = {
				value = 2
			}
		}
		# Much more likely with Technocracy
		if = {
			limit = {
				owner = {
					has_law = law_type:law_technocracy
				}
			}
			multiply = {
				value = 3
			}
		}
		# More likely with Appointed Bureaucrats
		if = {
			limit = {
				owner = {
					has_law = law_type:law_appointed_bureaucrats
				}
			}
			multiply = {
				value = 1.5
			}
		}
		# Impossible without Empiricism
		if = {
			limit = {
				owner = {
					NOT = { has_technology_researched = political_progress_1_cosmopolitan }
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	agitator_weight = {
		# Agitators more likely to come from populist IG's rather than elitist
		value = 0.75
		# Agitators will never be part of a marginalized IG
		if = {
			limit = {
				owner.ig:ig_urbanists = {
					is_marginal = yes
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	commander_weight = {
		value = 1.0
	}

	noble_chance = {
		value = 0.25
	}

	female_commander_chance = {
		value = 0.05
	}

	female_politician_chance = {
		value = 0.025

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}

			}
			add = {
				value = 0.05
			}
		}
	}

	female_agitator_chance = {
		value = 0.035
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}

			}
			add = {
				value = 0.035
			}
		}	
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}

			}
			add = {
				value = 0.035
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}

			}
			add = {
				value = 0.05
			}
		}		
	}

	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.2
		multiply = ig_commander_leader_chance_mult
	}
}
